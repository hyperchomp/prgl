36unsigned int prgl_create_shader(
37 const char *const vertex_source, const char *const frag_source,
38 const char *const geometry_source
39)
40{
41 // Create a basic vertex shader
42 unsigned int vertex_shader =
43 prgl_compile_shader(GL_VERTEX_SHADER, vertex_source);
44
45 // Create a fragment shader for color
46 unsigned int fragment_shader =
47 prgl_compile_shader(GL_FRAGMENT_SHADER, frag_source);
48
49 // Create geometry shader if one was passed in
50 int shader_count = 2;
51 unsigned int geometry_shader = 0;
52 if (geometry_source != NULL)
53 {
54 shader_count++;
55 geometry_shader =
56 prgl_compile_shader(GL_GEOMETRY_SHADER, geometry_source);
57 }
58
59 // Use a shader program to link the shaders together
60 unsigned int shaders[3] = {vertex_shader, fragment_shader, geometry_shader};
61 unsigned int shader_program =
62 prgl_create_shader_program(shaders, shader_count);
63
64 // Delete the shader objects, we're done with them
65 for (int i = 0; i < shader_count; i++)
66 {
67 glDeleteShader(shaders[i]);
68 }
69
70 return shader_program;
71}

176prgl_compile_shader(int gl_shader_type, const char *const shader_source)
177{
178 unsigned int shader = glCreateShader(gl_shader_type);
179 glShaderSource(shader, 1, &shader_source, NULL);

192static unsigned int
193prgl_create_shader_program(unsigned int shaders[], size_t length)
194{
195 unsigned int shader_program = glCreateProgram();
196 for (size_t i = 0; i < length; i++)

201 prgl_validate_shader_program(shader_program);
