48PRGLShader prgl_create_shader(
49 const char *const vertex_source[], int num_vertex_sources,
50 const char *const frag_source[], int num_frag_sources,
51 const char *const geometry_source[], int num_geometry_sources
52)
53{
54 // Create a basic vertex shader
55 GLuint vertex_shader = prgl_compile_shader(
56 GL_VERTEX_SHADER, vertex_source, num_vertex_sources
57 );
58
59 // Create a fragment shader for color
60 GLuint fragment_shader =
61 prgl_compile_shader(GL_FRAGMENT_SHADER, frag_source, num_frag_sources);
62
63 // Create geometry shader if one was passed in
64 int shader_count = 2;
65 GLuint geometry_shader = 0;
66 if (geometry_source != NULL)
67 {
68 shader_count++;
69 geometry_shader = prgl_compile_shader(
70 GL_GEOMETRY_SHADER, geometry_source, num_geometry_sources
71 );
72 }
73
74 // Use a shader program to link the shaders together
75 GLuint shaders[3] = {vertex_shader, fragment_shader, geometry_shader};
76 PRGLShader shader_program =
77 prgl_create_shader_program(shaders, shader_count);
78
79 // Delete the shader objects, we're done with them
80 for (int i = 0; i < shader_count; i++)
81 {
82 glDeleteShader(shaders[i]);
83 }
84
85 return shader_program;
86}
