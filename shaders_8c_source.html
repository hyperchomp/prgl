37unsigned int prgl_create_shader(
38 const char *const vertex_source, const char *const frag_source,
39 const char *const geometry_source
40)
41{
42 // Create a basic vertex shader
43 unsigned int vertex_shader =
44 prgl_compile_shader(GL_VERTEX_SHADER, vertex_source);
45
46 // Create a fragment shader for color
47 unsigned int fragment_shader =
48 prgl_compile_shader(GL_FRAGMENT_SHADER, frag_source);
49
50 // Create geometry shader if one was passed in
51 int shader_count = 2;
52 unsigned int geometry_shader = 0;
53 if (geometry_source != NULL)
54 {
55 shader_count++;
56 geometry_shader =
57 prgl_compile_shader(GL_GEOMETRY_SHADER, geometry_source);
58 }
59
60 // Use a shader program to link the shaders together
61 unsigned int shaders[3] = {vertex_shader, fragment_shader, geometry_shader};
62 unsigned int shader_program =
63 prgl_create_shader_program(shaders, shader_count);
64
65 // Delete the shader objects, we're done with them
66 for (int i = 0; i < shader_count; i++)
67 {
68 glDeleteShader(shaders[i]);
69 }
70
71 return shader_program;
72}

204{
205 unsigned int shader_program = glCreateProgram();
206 for (size_t i = 0; i < length; i++)
