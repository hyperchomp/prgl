277void pr3d_calculate_aabb(
278 const float *vertices, int num_vertices, vec3 min_bounds, vec3 max_bounds
279)
280{
281 // Initialize bounds using first vertex
282 min_bounds[0] = vertices[0];
283 min_bounds[1] = vertices[1];
284 min_bounds[2] = vertices[2];
285
286 max_bounds[0] = vertices[0];
287 max_bounds[1] = vertices[1];
288 max_bounds[2] = vertices[2];
289
290 for (int i = 0; i < num_vertices; i++)
291 {
292 // Find min bounds from the points in the vertices
293 for (int p = 0; p < 3; p++)
294 {
295 if (vertices[i * 3 + p] < min_bounds[p])
296 {
297 min_bounds[p] = vertices[i * 3 + p];
298 }
299 }
300
301 // Find max bounds
302 for (int p = 0; p < 3; p++)
303 {
304 if (vertices[i * 3 + p] > max_bounds[p])
305 {
306 max_bounds[p] = vertices[i * 3 + p];
307 }
308 }
309 }
