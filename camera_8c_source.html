16void prgl_init_camera(
17 struct PRGLCamera *const cam, float fov_degrees, float move_speed,
18 enum PRGLCameraProjectionType projection_type
19)
20{
21 prgl_active_camera_ref = cam;
22
23 glm_vec3_zero(cam->position);
24 glm_vec3_zero(cam->up);
25 glm_vec3_zero(cam->front);
26 glm_vec3_zero(cam->right);
27
28 cam->up[1] = 1.0f; // Up is +y
29 cam->front[2] = -1.0f; // Forward is -z
30
31 // With angles 0 degrees is pointing straight to the right. For yaw we are
32 // looking down the y axis, so in a graph view we are looking at horizontal
33 // x and vertical z. Our forward axis is -z, so we need to rotate -90
34 // degrees to be looking "forward".
35 cam->yaw = -90.0f;
36

38 cam->move_speed = move_speed;
39 cam->look_sensitivity = 0.1f;

44void prgl_update_camera(struct PRGLCamera *const cam)
45{
46 // Calculate front vector, this is the z-axis
47 vec3 front;
48 const float yaw_rads = glm_rad(cam->yaw);
49 const float pitch_rads = glm_rad(cam->pitch);
50 const float cos_pitch = cosf(pitch_rads);
51 front[0] = cosf(yaw_rads) * cos_pitch;
52 front[1] = sinf(pitch_rads);
53 front[2] = sinf(yaw_rads) * cos_pitch;
54 glm_normalize(front);
55 glm_vec3_copy(front, cam->front);
56
57 // Right and Up vector as well
58 glm_cross(cam->front, PRGL_WORLD_UP, cam->right); // This is the x-axis
59 glm_normalize(cam->right);
60
61 glm_cross(cam->right, cam->front, cam->up); // This is the y-axis
62 glm_normalize(cam->up);
63
64 // Now calculate the direction of the camera, and adjust the view matrix
65 vec3 direction;
66 glm_vec3_add(cam->position, cam->front, direction);
67 glm_lookat(cam->position, direction, cam->up, cam->view);
68
69 // The view matrix is the camera position, which we calculated above
70 prgl_set_shader_uniform_mat4(prgl_current_shader(), "view", cam->view);
71}

73void prgl_move_camera_fly(
74 struct PRGLCamera *const cam, enum PRGLCameraMoveDirection move_dir,
75 double delta_time
76)
77{
78 const float velocity = cam->move_speed * (float)delta_time;
79 vec3 front_velocity;
80 glm_vec3_scale(cam->front, velocity, front_velocity);
81 vec3 right_velocity;
82 glm_vec3_scale(cam->right, velocity, right_velocity);
83
84 switch (move_dir)
85 {
86 case PRGL_CAMERA_MOVE_DIR_FORWARD:
87 glm_vec3_add(cam->position, front_velocity, cam->position);
88 break;
89 case PRGL_CAMERA_MOVE_DIR_BACKWARD:
90 glm_vec3_sub(cam->position, front_velocity, cam->position);
91 break;
92 case PRGL_CAMERA_MOVE_DIR_LEFT:
93 glm_vec3_sub(cam->position, right_velocity, cam->position);
94 break;
95 case PRGL_CAMERA_MOVE_DIR_RIGHT:
96 glm_vec3_add(cam->position, right_velocity, cam->position);
97 break;
98 default:
99 fprintf(
100 stderr,
101 "prgl_move_camera: Unspecified camera move direction type\n"
102 );
103 exit(EXIT_FAILURE);
104 }
105}

119void prgl_set_camera_projection(
120 struct PRGLCamera *const cam, float fov_degrees,
121 enum PRGLCameraProjectionType projection_type
122)
123{
124 struct PRGLScreen *const screen = prgl_screen();
125 if (projection_type == PRGL_CAMERA_PROJECTION_PERSPECTIVE)
126 {
127 // Perspective needs fov in radians, aspect, and near/far clip
128 cam->fov = fov_degrees;
129 const float fov = glm_rad(fov_degrees);
130
131 const float aspect_ratio = screen->aspect_ratio;
132
133 glm_perspective(
134 fov, aspect_ratio, 0.1f, 100.0f, cam->projection_perspective
135 );
136 prgl_set_shader_uniform_mat4(
137 prgl_current_shader(), "projection", cam->projection_perspective
138 );
139 }
140 else if (projection_type == PRGL_CAMERA_PROJECTION_ORTHOGONAL)
141 {
142 // Orthogonal is 2D so uses width/height, the clipping plane is the
143 // OpenGL coordinate plane which goes from -1.0 to 1.0
144 glm_ortho(
145 0.0f, PRGL_RENDER_RESOLUTION[0], PRGL_RENDER_RESOLUTION[1], 0.0f,
146 -1.0f, 1.0f, cam->projection_orthogonal
147 );
148 prgl_set_shader_uniform_mat4(
149 prgl_current_shader(), "projection", cam->projection_orthogonal
