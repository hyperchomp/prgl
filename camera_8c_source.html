16void pr3d_init_camera(
17 struct PR3DCamera *cam, float fov_degrees, float move_speed,
18 enum PR3DCameraProjectionType projection_type
19)
20{
21 pr3d_active_camera_ref = cam;
22
23 glm_vec3_zero(cam->position);
24 glm_vec3_zero(cam->up);
25 glm_vec3_zero(cam->front);

27
28 cam->up[1] = 1.0f; // Up is +y

31 // With angles 0 degrees is pointing straight to the right. For yaw we are

34 // degrees to be looking "forward".

44void pr3d_update_camera(struct PR3DCamera *cam)
45{
46 // Calculate front vector, this is the z-axis
47 vec3 front;
48 const float yaw_rads = glm_rad(cam->yaw);
49 const float pitch_rads = glm_rad(cam->pitch);
50 const float cos_pitch = cosf(pitch_rads);
51 front[0] = cosf(yaw_rads) * cos_pitch;
52 front[1] = sinf(pitch_rads);
53 front[2] = sinf(yaw_rads) * cos_pitch;
54 glm_normalize(front);
55 glm_vec3_copy(front, cam->front);
56
57 // Right and Up vector as well
58 glm_cross(cam->front, PR3D_WORLD_UP, cam->right); // This is the x-axis
59 glm_normalize(cam->right);
60
61 glm_cross(cam->right, cam->front, cam->up); // This is the y-axis
62 glm_normalize(cam->up);
63
64 // Now calculate the direction of the camera, and adjust the view matrix
65 vec3 direction;
66 glm_vec3_add(cam->position, cam->front, direction);
67 glm_lookat(cam->position, direction, cam->up, cam->view);
68
69 // The view matrix is the camera position, which we calculated above
70 pr3d_set_shader_uniform_mat4(pr3d_current_shader(), "view", cam->view);
71}

73void pr3d_move_camera_fly(
74 struct PR3DCamera *cam, const enum PR3DCameraMoveDirection move_dir,
75 const double delta_time
76)
77{
78 const float velocity = cam->move_speed * (float)delta_time;
79 vec3 front_velocity;
80 glm_vec3_scale(cam->front, velocity, front_velocity);
81 vec3 right_velocity;
82 glm_vec3_scale(cam->right, velocity, right_velocity);
83
84 switch (move_dir)
85 {
86 case PR3D_CAMERA_MOVE_DIR_FORWARD:
87 glm_vec3_add(cam->position, front_velocity, cam->position);
88 break;
89 case PR3D_CAMERA_MOVE_DIR_BACKWARD:
90 glm_vec3_sub(cam->position, front_velocity, cam->position);
91 break;
92 case PR3D_CAMERA_MOVE_DIR_LEFT:
93 glm_vec3_sub(cam->position, right_velocity, cam->position);
94 break;
95 case PR3D_CAMERA_MOVE_DIR_RIGHT:
96 glm_vec3_add(cam->position, right_velocity, cam->position);
97 break;
98 default:
99 fprintf(
100 stderr,
101 "pr3d_move_camera: Unspecified camera move direction type\n"
102 );
103 exit(EXIT_FAILURE);
104 }
105}

136 pr3d_set_shader_uniform_mat4(
137 pr3d_current_shader(), "projection", cam->projection_perspective
138 );
139 }
140 else if (projection_type == PR3D_CAMERA_PROJECTION_ORTHOGONAL)
141 {
142 // Orthogonal is 2D so uses width/height, the clipping plane is the

144 glm_ortho(
