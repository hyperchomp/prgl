283void pr3d_render_mesh(
284 struct PR3DMesh *mesh, vec3 position, vec3 rotation_axis, float degrees,
285 vec3 scale
286)
287{
288 glBindVertexArray(mesh->vao);
289
290 if (mesh->texture != 0)
291 {
292 glBindTexture(GL_TEXTURE_2D, mesh->texture);
293 }
294
295 // Transform the mesh to the render position.
296 mat4 trans;
297 glm_mat4_identity(trans);
298 glm_translate(trans, position);
299 glm_rotate(trans, glm_rad(degrees), rotation_axis);
300 glm_scale(trans, scale);
301 pr3d_set_shader_uniform_mat4(
302 pr3d_current_shader(), PR3D_TRANSFORM_UNIFORM, trans
303 );
304
305 if (mesh->ebo == 0)
306 {
307 glDrawArrays(GL_TRIANGLES, 0, mesh->num_vertices);
308 }
309 else
310 {
311 glDrawElements(GL_TRIANGLES, mesh->num_vertices, GL_UNSIGNED_INT, 0);
312 }
313}
